## 项目1: 研究助手Agent

### 🎯 项目目标
构建一个能够自动搜索、总结和保存信息的研究助手。

### 📝 功能需求
1. 接收研究主题
2. 搜索相关资料（Wikipedia + 网络搜索）
3. 总结关键信息
4. 保存研究报告到文件

### 🛠️ 技术要点
- Agent创建和配置
- 多工具集成（搜索、文件操作）
- 信息整合和总结
- 结果格式化

### 📖 实现步骤

#### 步骤1: 准备工具
```python
from langchain.agents import Tool
from tools.search import search_wikipedia, search_duckduckgo
import os

def save_report(content: str) -> str:
    """保存研究报告"""
    filename = f"research_report_{datetime.now().strftime('%Y%m%d_%H%M%S')}.md"
    os.makedirs('reports', exist_ok=True)
    filepath = os.path.join('reports', filename)

    with open(filepath, 'w', encoding='utf-8') as f:
        f.write(content)

    return f"✅ 报告已保存到: {filepath}"

tools = [
    Tool(
        name="WikipediaSearch",
        func=search_wikipedia,
        description="搜索Wikipedia获取权威信息"
    ),
    Tool(
        name="WebSearch",
        func=search_duckduckgo,
        description="搜索互联网获取最新信息"
    ),
    Tool(
        name="SaveReport",
        func=save_report,
        description="保存研究报告到文件。输入应该是完整的报告内容。"
    )
]
```

## 项目2: 文档处理Workflow

### 🎯 项目目标
构建一个自动化的文档处理流程：上传→解析→摘要→翻译→保存。

### 📝 功能需求
1. 读取文档内容
2. 提取关键信息
3. 生成中文摘要
4. 翻译成英文
5. 保存处理结果

### 🛠️ 技术要点
- SequentialChain使用
- 多步骤流程设计
- 数据流转
- 错误处理

### 📖 实现步骤

#### 步骤1: 定义处理步骤
```python
from langchain_openai import ChatOpenAI
from langchain.chains import LLMChain, SequentialChain
from langchain.prompts import PromptTemplate

llm = ChatOpenAI(model="gpt-4o-mini", temperature=0.7)

# 步骤1: 提取关键信息
extract_prompt = PromptTemplate(
    input_variables=["document"],
    template="""从以下文档中提取关键信息:

文档内容:
{document}

请提取:
1. 主题
2. 关键人物/组织
3. 重要日期
4. 核心观点

关键信息:"""
)

extract_chain = LLMChain(
    llm=llm,
    prompt=extract_prompt,
    output_key="key_info"
)

# 步骤2: 生成摘要
summary_prompt = PromptTemplate(
    input_variables=["key_info"],
    template="""基于以下关键信息，生成一个200字左右的中文摘要:

关键信息:
{key_info}

摘要:"""
)

summary_chain = LLMChain(
    llm=llm,
    prompt=summary_prompt,
    output_key="summary_cn"
)

# 步骤3: 翻译成英文
translate_prompt = PromptTemplate(
    input_variables=["summary_cn"],
    template="""将以下中文摘要翻译成英文:

中文摘要:
{summary_cn}

英文翻译:"""
)

translate_chain = LLMChain(
    llm=llm,
    prompt=translate_prompt,
    output_key="summary_en"
)

# 步骤4: 格式化输出
format_prompt = PromptTemplate(
    input_variables=["key_info", "summary_cn", "summary_en"],
    template="""生成最终报告:

# 文档处理报告

## 关键信息
{key_info}

## 中文摘要
{summary_cn}

## English Summary
{summary_en}

---
生成时间: {timestamp}
"""
)

format_chain = LLMChain(
    llm=llm,
    prompt=format_prompt,
    output_key="final_report"
)
```

#### 步骤2: 组合Workflow
```python
from datetime import datetime

workflow = SequentialChain(
    chains=[
        extract_chain,
        summary_chain,
        translate_chain,
        format_chain
    ],
    input_variables=["document"],
    output_variables=["key_info", "summary_cn", "summary_en", "final_report"],
    verbose=True
)
```

#### 步骤3: 处理文档
```python
def process_document(filepath: str) -> dict:
    """处理文档"""

    # 读取文档
    with open(filepath, 'r', encoding='utf-8') as f:
        document = f.read()

    # 执行workflow
    result = workflow.invoke({
        "document": document,
        "timestamp": datetime.now().strftime('%Y-%m-%d %H:%M:%S')
    })

    # 保存结果
    output_path = filepath.replace('.txt', '_processed.md')
    with open(output_path, 'w', encoding='utf-8') as f:
        f.write(result['final_report'])

    print(f"✅ 处理完成，结果保存到: {output_path}")

    return result
```

#### 步骤4: 批量处理
```python
import os

def batch_process(directory: str):
    """批量处理目录下的所有文档"""

    for filename in os.listdir(directory):
        if filename.endswith('.txt'):
            filepath = os.path.join(directory, filename)
            print(f"\n处理文件: {filename}")

            try:
                process_document(filepath)
            except Exception as e:
                print(f"❌ 处理失败: {e}")
```

#### 步骤1: 准备知识库
```python
# 简单的FAQ知识库
FAQ_DATABASE = {
    "退款": "退款流程：1. 进入订单页面 2. 选择要退款的订单 3. 点击申请退款 4. 填写退款原因 5. 等待审核（1-3个工作日）",
    "发货": "发货时间：工作日下单当天发货，周末下单周一发货。物流信息可在订单详情中查看。",
    "支付": "支持的支付方式：微信支付、支付宝、银行卡。支付遇到问题请联系客服。",
    "会员": "会员权益：1. 专属折扣 2. 优先发货 3. 专属客服 4. 积分翻倍",
}

def search_faq(query: str) -> str:
    """搜索FAQ"""
    query_lower = query.lower()

    for key, value in FAQ_DATABASE.items():
        if key in query_lower:
            return f"📚 找到相关信息:\n{value}"

    return "❌ 未找到相关FAQ，请联系人工客服"
```

def query_order(order_id: str) -> str:
    """查询订单状态（模拟）"""
    # 实际应该查询数据库
    mock_orders = {
        "12345": {"status": "已发货", "tracking": "SF1234567890"},
        "67890": {"status": "待发货", "tracking": None},
    }

    if order_id in mock_orders:
        order = mock_orders[order_id]
        if order['tracking']:
            return f"订单{order_id}状态: {order['status']}\n物流单号: {order['tracking']}"
        else:
            return f"订单{order_id}状态: {order['status']}"
    else:
        return f"未找到订单{order_id}"

def apply_refund(order_id: str, reason: str) -> str:

#### 步骤3: 创建客服Agent
```python
from langchain_openai import ChatOpenAI
from langchain.agents import AgentExecutor, create_react_agent, Tool
from langchain.memory import ConversationBufferWindowMemory
from langchain.prompts import PromptTemplate

# 初始化
llm = ChatOpenAI(model="gpt-4o-mini", temperature=0.7)

# 定义工具
tools = [
    Tool(
        name="SearchFAQ",
        func=search_faq,
        description="搜索常见问题FAQ。输入用户的问题关键词。"
    ),
    Tool(
        name="QueryOrder",
        func=query_order,
        description="查询订单状态。输入订单号。"
    ),
    Tool(
        name="ApplyRefund",
        func=apply_refund,
        description="申请退款。输入格式: '订单号|退款原因'"
    ),
    Tool(
        name="CheckPoints",
        func=check_points,
        description="查询用户积分。输入用户ID。"
    )
]

# 创建Memory
memory = ConversationBufferWindowMemory(
    k=5,
    memory_key="chat_history",
    return_messages=True
)

# 自定义Prompt
prompt = PromptTemplate.from_template("""
你是一个友好、专业的客服助手。

你的职责:
1. 理解用户问题
2. 使用合适的工具帮助用户
3. 提供清晰、友好的回答
4. 记住对话历史

可用工具:
{tools}

工具名称: {tool_names}

对话历史:
{chat_history}

用户: {input}
{agent_scratchpad}
""")

# 创建Agent
agent = create_react_agent(llm, tools, prompt)

agent_executor = AgentExecutor(
    agent=agent,
    tools=tools,
    memory=memory,
    verbose=True,
    max_iterations=5,
    handle_parsing_errors=True
)
```

#### 步骤4: 交互界面
```python
def customer_service_chat():
    """客服聊天界面"""

    print("=" * 60)
    print("🤖 智能客服机器人")
    print("=" * 60)
    print("\n我可以帮您:")
    print("  • 查询订单状态")
    print("  • 申请退款")
    print("  • 查询积分")
    print("  • 解答常见问题")
    print("\n输入 'quit' 退出")
    print("=" * 60)

    while True:
        try:
            user_input = input("\n您: ").strip()

            if user_input.lower() in ['quit', 'exit', '退出']:
                print("\n👋 感谢使用，再见！")
                break

            if not user_input:
                continue

            print("\n客服: ", end="", flush=True)

            result = agent_executor.invoke({"input": user_input})
            print(result['output'])

        except KeyboardInterrupt:
            print("\n\n👋 再见！")
            break
        except Exception as e:
            print(f"\n❌ 抱歉，出现了错误: {e}")
            print("请重新描述您的问题。")

# 运行
if __name__ == "__main__":
    customer_service_chat()
```
